<?php
class PremiseVersion {

    public function __construct() {
        add_action( 'admin_menu', array( $this, 'add_plugin_page' ) );
        add_action( 'admin_init', array( $this, 'page_init' ) );
    }

    public function add_plugin_page() {
        add_options_page('Premise Demo', 'Premise Demo', 'manage_options', 'premise-demo', array( $this, 'create_admin_page' ));
    }

    public function page_init() {
        register_setting('premisedemo','premisedemo');
    }

    public function create_admin_page() { ?>
        <div class="wrap">
            <h2>Premise Demo Page</h2>
            <p>Welcome to our Wordpress demo. We'll try to illustrate how simple Premise WP really is. Check out what happens on the screen to the right, and see how this page changes.</p>
            <div class="premise-clear" style="padding-bottom: 20px;"></div><form method="post" class="span8" action="options.php">
            <?php
                // Register your settings from
                // 'page_init()' above.
                settings_fields( 'premisedemo' );

                premise_field('text', array(
                    'name'=>'premisedemo[title]',
                    'label' => 'Ttile',
                    'tooltip' => 'Lorem ipsum dolor sit amet, consectetur adipisicing elit, sed do eiusmod tempor incididunt ut labore et dolore magna aliqua. Ut enim ad minim veniam, quis nostrud exercitation ullamco laboris nisi ut aliquip ex ea commodo consequat. Excepteur sint occaecat cupidatat non proident, sunt in culpa qui officia deserunt mollit anim id est laborum.',
                ));

                premise_field('fa_icon', array(
                    'name'=>'premisedemo[icon]',
                    'label' => 'Icon'
                ));

                premise_field('textarea', array(
                    'name'=>'premisedemo[message]',
                    'label' => 'Message'
                ));

                premise_field('wp_media', array(
                    'name'=>'premisedemo[image]',
                    'label' => 'Media'
                ));

                premise_field('submit', array('class'=>'premise-float-right'));

            ?>
            </form></div>
        <?php
    }

    
}
